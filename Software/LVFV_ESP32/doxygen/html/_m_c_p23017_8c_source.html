111static esp_err_t i2c_master_init(void) {
112 i2c_config_t conf = {
113 .mode = I2C_MODE_MASTER,
114 .sda_io_num = I2C_MASTER_SDA_IO,
115 .scl_io_num = I2C_MASTER_SCL_IO,
116 .sda_pullup_en = GPIO_PULLUP_ENABLE,
117 .scl_pullup_en = GPIO_PULLUP_ENABLE,
118 .master.clk_speed = I2C_MASTER_FREQ_HZ,
119 };
120 ESP_ERROR_CHECK(i2c_param_config(I2C_IO_MASTER_NUM, &conf));
121 ESP_ERROR_CHECK(i2c_driver_install(I2C_IO_MASTER_NUM, conf.mode,
122 I2C_MASTER_RX_BUF_DISABLE,
123 I2C_MASTER_TX_BUF_DISABLE, 0));
124 if (xI2C_Mutex == NULL) {
125 xI2C_Mutex = xSemaphoreCreateMutex();
126 if (xI2C_Mutex == NULL) {
127 ESP_LOGE(TAG, "No se pudo crear el mutex de I2C");
128 }
129 }
130
131 return ESP_OK;
132}

134static bool i2c_is_hardware_free() {
135 while ( xSemaphoreTake(xI2C_Mutex, pdMS_TO_TICKS(100)) != pdTRUE ) {
136 ESP_LOGI(TAG, "Espero recursos de hardware I2C...");
137 return false;
138 }
139 return true;
140}

146static esp_err_t mcp_register_read(uint8_t register_address, uint8_t *data) {
147 esp_err_t ret = ESP_OK;
148 i2c_cmd_handle_t cmd = i2c_cmd_link_create();
149 i2c_master_start(cmd);
150 i2c_master_write_byte(cmd, __MCP23017_WRITE_OPPCODE__,0);
151 i2c_master_write_byte(cmd, register_address, true);
152 i2c_master_start(cmd);
153 i2c_master_write_byte(cmd, __MCP23017_READ_OPPCODE__,0);
154 i2c_master_read_byte(cmd, data, I2C_MASTER_NACK);
155 i2c_master_stop(cmd);
156 ret = i2c_master_cmd_begin(I2C_IO_MASTER_NUM, cmd, pdMS_TO_TICKS(I2C_MASTER_TIMEOUT_MS));
157 i2c_cmd_link_delete(cmd);
158 return ret;
159}
