357esp_err_t set_freq_output(uint16_t freq) {
358 esp_err_t retval;
359 if (freq > 450) {
360 freq = 450;
361 }
362
363 // Calcular duty linealmente (en %)
364 float duty_percent = -0.0977f * freq + 98.0f;
365
366 // Convertir duty en ticks de PWM (0..1023)
367 uint32_t duty = (uint32_t)((duty_percent / 100.0f) * 1023.0f);
368 if ( ledc_get_duty(PWM_MODE, PWM_CHANNEL) != duty ) {
369 retval = ledc_set_duty(PWM_MODE, PWM_CHANNEL, duty);
370 if ( retval != ESP_OK ) {
371 return retval;
372 }
373 retval = ledc_update_duty(PWM_MODE, PWM_CHANNEL);
374 if ( retval != ESP_OK ) {
375 return retval;
376 }
377 ESP_LOGI("PWM", "Freq = %dHz, Duty = %f%%", freq, duty_percent);
378 }
379 return ESP_OK;
380}

382void GPIO_interrupt_attendance_task(void* arg) {
383 uint32_t io_num;
384 uint32_t last_interrupt = 0;
385 uint8_t scratch_data;
386 uint8_t speed_selector;
387
388 if (MCP23017_INIT() == ESP_OK ) {
389 ESP_LOGI(TAG, "MCP23017 Tarea iniciada correctamente");
390 } else {
391 ESP_LOGE(TAG, "Error al inicializar el MCP23017");
392 ESP_ERROR_CHECK(ESP_FAIL);
393 }
394
395 gpio_init_interrupts();
396 freq_0_10V_output();
397 set_freq_output(0);
398
399 xTaskCreate(MCP23017_buzzer_control, "MCP23017_buzzer_control", 2048, NULL, 10, NULL);
400 xTaskCreate(MCP23017_relay_control, "MCP23017_relay_control", 2048, NULL, 10, NULL);
401 xTaskCreate(MCP23017_keyboard_control, "MCP23017_keyboard_control", 2048, NULL, 10, NULL);
402
403 vTaskDelay(pdMS_TO_TICKS(100));
404
405 for(;;) {
406 system_status_t s_e;
407 get_status(&s_e);
408 set_freq_output(s_e.frequency);
409 if(xQueueReceive(GPIO_evt_queue, &io_num, pdMS_TO_TICKS(200))) {
410 if ( io_num == INT_A_PIN ) {
411 ESP_ERROR_CHECK_WITHOUT_ABORT( mcp_interrupt_flag(PORTA, &scratch_data) );
412 ESP_ERROR_CHECK_WITHOUT_ABORT( mcp_get_on_interrupt_input(PORTA, &mcp_porta) );
413
414 if ( scratch_data & TERMO_SW_MASK ) {
415 if ( mcp_porta & TERMO_SW_MASK ) {
416 last_interrupt = TERMO_SW_RELEASED;
417 } else {
418 last_interrupt = TERMO_SW_PRESSED;
419 }
420 } else if ( scratch_data & FREQ_SEL_MASK ) {
421 mcp_porta = ~mcp_porta;
422 speed_selector = mcp_porta & FREQ_SEL_MASK; // 00ss ss00
423 speed_selector = ( speed_selector >> ( FREQ_SEL_1_PIN ) ); // 0000 ssss
424 last_interrupt = SPEED_SELECTOR_0 + speed_selector;
425 } else if ( scratch_data & STOP_SW_MASK ) {
426 if ( mcp_porta & STOP_SW_MASK ) {
427 last_interrupt = STOP_RELEASED;
428 } else {
429 last_interrupt = STOP_PRESSED;
430 }
431 }
432 } else if ( io_num == INT_B_PIN ) {
433 ESP_ERROR_CHECK_WITHOUT_ABORT( mcp_read_port(PORTB, &mcp_portb) );
434 switch (mcp_portb){
435 case SWITCH_BUTTON_MENU:
436 last_interrupt = BUTTON_MENU;
437 break;
438 case SWITCH_BUTTON_DOWN:
439 last_interrupt = BUTTON_DOWN;
440 break;
441 case SWITCH_BUTTON_RIGHT:
442 last_interrupt = BUTTON_RIGHT;
443 break;
444 case SWITCH_BUTTON_LEFT:
445 last_interrupt = BUTTON_LEFT;
446 break;
447 case SWITCH_BUTTON_UP:
448 last_interrupt = BUTTON_UP;
449 break;
450 case SWITCH_BUTTON_OK:
451 last_interrupt = BUTTON_OK;
452 break;
453 case SWITCH_BUTTON_SAVE:
454 last_interrupt = BUTTON_SAVE;
455 break;
456 case SWITCH_BUTTON_BACK:
457 last_interrupt = BUTTON_BACK;
458 break;
459 }
460 } else if ( io_num == STOP_BUTTON_PIN ) {
461 if ( !gpio_get_level(io_num) ) {
462 last_interrupt = EMERGENCI_STOP_PRESSED;
463 } else {
464 last_interrupt = EMERGENCI_STOP_RELEASED;
465 }
466 } else if ( io_num == START_BUTTON_PIN ) {
467 if ( !gpio_get_level(io_num) ) {
468 ESP_LOGI(TAG, "Start button pressed");
469 last_interrupt = START_PRESSED;
470 } else {
471 ESP_LOGI(TAG, "Start button released");
472 last_interrupt = START_RELEASED;
473 }
474 }
475 continue;
476 }
477 switch (last_interrupt) {
478 case BUTTON_MENU:
479 BuzzerEvantPost( 50 );
480 break;
481 case BUTTON_DOWN:
482 BuzzerEvantPost( 50 );
483 break;
484 case BUTTON_RIGHT:
485 BuzzerEvantPost( 50 );
486 break;
487 case BUTTON_LEFT:
488 BuzzerEvantPost( 50 );
489 break;
490 case BUTTON_UP:
491 BuzzerEvantPost( 50 );
492 break;
493 case BUTTON_OK:
494 BuzzerEvantPost( 50 );
495 break;
496 case BUTTON_SAVE:
497 BuzzerEvantPost( 50 );
498 break;
499 case BUTTON_BACK:
500 BuzzerEvantPost( 50 );
501 break;
502 case TERMO_SW_PRESSED:
503 RelayEvantPost( 1 );
504 break;
505 case TERMO_SW_RELEASED:
506 RelayEvantPost( 0 );
507 break;
508 case STOP_PRESSED:
509 RelayEvantPost( 1 );
510 break;
511 case STOP_RELEASED:
512 RelayEvantPost( 0 );
513 break;
514 case SPEED_SELECTOR_0:
515 case SPEED_SELECTOR_1:
516 case SPEED_SELECTOR_2:
517 case SPEED_SELECTOR_3:
518 case SPEED_SELECTOR_4:
519 case SPEED_SELECTOR_5:
520 case SPEED_SELECTOR_6:
521 case SPEED_SELECTOR_7:
522 case SPEED_SELECTOR_8:
523 case SPEED_SELECTOR_9:
524 ESP_LOGI(TAG, "Pulsado speed_slector %d", last_interrupt);
525 case EMERGENCI_STOP_PRESSED:
526 case EMERGENCI_STOP_RELEASED:
527 case START_PRESSED:
528 case START_RELEASED:
529 break;
530 }
531 DisplayEventPost(last_interrupt);
532 last_interrupt = 0;
533 }
534}
