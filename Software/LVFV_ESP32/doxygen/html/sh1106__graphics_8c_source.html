199static void sh1106_draw_bitmap( sh1106_t *oled, bitmap_t *bitmap) {
200 // Dibujado sencillo: copia bit por bit en el buffer
201 // (Implementar según formato del bitmap)
202 for (uint8_t by = 0; by < bitmap->h; by++) {
203 for (uint8_t bx = 0; bx < bitmap->w; bx++) {
204 uint8_t byte = bitmap->bitmap[by * ((bitmap->w + 7) / 8) + (bx / 8)];
205 uint8_t bit = (byte >> (7 - (bx % 8))) & 0x01;
206 if (bit) {
207 int px = bitmap->x + bx;
208 int py = bitmap->y + by;
209 if (px >= 0 && px < oled->width && py >= 0 && py < oled->height) {
210 uint8_t page = py / 8;
211 uint8_t bitpos = py % 8;
212 oled->buffer[page * oled->width + px] |= (1 << bitpos);
213 }
214 }
215 }
216 }
