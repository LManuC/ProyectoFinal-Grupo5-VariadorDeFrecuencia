204static esp_err_t sh1106_refresh() {
205 uint8_t page;
206 // El SH1106 espera que se setee la dirección de la columna luego se manden datos página por página
207 for ( page = 0; page < 8; page++) {
208 // Dirección de página
209 uint8_t command = 0xB0 + page;
210 ESP_ERROR_CHECK(sh1106_write(&command, 1, SH1106_COMM_CMD));
211 // Dirección de columna baja y alta (offset 2 que ajusta SH1106)
212 command = 0x02;
213 ESP_ERROR_CHECK(sh1106_write(&command, 1, SH1106_COMM_CMD)); // col lo (2)
214 command = 0x10;
215 ESP_ERROR_CHECK(sh1106_write(&command, 1, SH1106_COMM_CMD)); // col hi
216
217 // Enviar 128 bytes de datos para esta página
218 uint8_t *page_data = &oled.buffer[page * 128];
219 ESP_ERROR_CHECK(sh1106_write(page_data, 128, SH1106_COMM_DATA));
220 }
221 if ( page == 8 ) {
222 return ESP_OK;
223 }
224 return ESP_FAIL;
225}

466esp_err_t sh1106_init() {
467
468 oled.width = 128;
469 oled.height = 64;
470 oled.rotation = 0;
471
472 i2c_master_init();
473
474 for (size_t i = 0; i < sizeof(init_seq); i++) {
475 if ( sh1106_write(&init_seq[i], 1, SH1106_COMM_CMD) != ESP_OK ) {
476 ESP_LOGE(TAG, "Error enviando comando de inicialización %zu: 0x%02X", i, init_seq[i]);
477 return ESP_FAIL;
478 }
479 }
480 sh1106_clear_buffer();
481 sh1106_splash_screen();
482 return sh1106_refresh();
483}
