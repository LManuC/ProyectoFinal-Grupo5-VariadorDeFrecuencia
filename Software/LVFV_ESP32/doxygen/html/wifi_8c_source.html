91static int hex2int(char c) {
92 if (c >= '0' && c <= '9') return c - '0';
93 if (c >= 'a' && c <= 'f') return 10 + (c - 'a');
94 if (c >= 'A' && c <= 'F') return 10 + (c - 'A');
95 return 0;
96}

98static void url_decode(char *dst, const char *src) {
99 while (*src) {
100 if (*src == '+') {
101 *dst++ = ' ';
102 src++;
103 } else if (*src == '%' && isxdigit((unsigned char)src[1]) && isxdigit((unsigned char)src[2])) {
104 int hi = hex2int(src[1]);
105 int lo = hex2int(src[2]);
106 *dst++ = (char)((hi << 4) | lo);
107 src += 3;
108 } else {
109 *dst++ = *src++;
110 }
111 }
112 *dst = '\0';
113}

115static bool get_param_value(const char *body, const char *key, char *dest, size_t dest_len) {
116 size_t key_len = strlen(key);
117 const char *p = body;
118
119 while (p && *p) {
120 const char *eq = strchr(p, '=');
121 if (!eq) break;
122
123 // Nombre de parámetro
124 size_t this_key_len = (size_t)(eq - p);
125
126 if (this_key_len == key_len && strncmp(p, key, key_len) == 0) {
127 // Encontramos la clave. Buscar fin del valor (& o final de cadena)
128 const char *val_start = eq + 1;
129 const char *amp = strchr(val_start, '&');
130 size_t val_len = amp ? (size_t)(amp - val_start) : strlen(val_start);
131 if (val_len >= dest_len) val_len = dest_len - 1;
132
133 char encoded[256];
134 if (val_len >= sizeof(encoded)) val_len = sizeof(encoded) - 1;
135
136 memcpy(encoded, val_start, val_len);
137 encoded[val_len] = '\0';
138
139 url_decode(dest, encoded);
140 return true;
141 }
142
143 // Ir al siguiente par key=value
144 const char *amp = strchr(eq + 1, '&');
145 if (!amp) break;
146 p = amp + 1;
147 }
148
149 return false;
150}
