104static esp_err_t save_16 (const char *var_tag, int16_t value) {
105 nvs_handle_t h;
106 esp_err_t err;
107
108 err = nvs_open("storage", NVS_READWRITE, &h);
109 if (err != ESP_OK) {
110 return err;
111 }
112
113 ESP_LOGI(TAG, "Guardando %s = %d", var_tag, (int)value);
114 err = nvs_set_i16(h, var_tag, value);
115 if (err == ESP_OK) err = nvs_commit(h);
116
117 nvs_close(h);
118 return err;
119}

203static esp_err_t load_16(const char *var_tag, int16_t *value, int16_t defval) {
204 nvs_handle_t h;
205
206 esp_err_t err;
207
208 if ( var_tag == NULL ) {
209 return ESP_FAIL;
210 }
211
212 err = nvs_open("storage", NVS_READWRITE, &h);
213
214 if (err != ESP_OK) {
215 return err;
216 }
217
218 if (err != ESP_OK) {
219 *value = defval;
220 return err;
221 }
222
223 err = nvs_get_i16(h, var_tag, value);
224 if (err == ESP_ERR_NVS_NOT_FOUND) {
225 *value = defval;
226 ESP_LOGW(TAG, "Clave %s no encontrada, usando default=%d", var_tag, (int)defval);
227 err = ESP_OK;
228 } else if (err == ESP_OK) {
229 ESP_LOGI(TAG, "Cargado %s = %d", var_tag, (int)*value); // <-- *value
230 }
231
232 nvs_close(h);
233 return err;
234}
