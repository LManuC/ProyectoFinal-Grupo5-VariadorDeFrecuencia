162 vTaskDelete( desaccelerating_handle );
163 desaccelerating_handle = NULL;
164 }

166 system_status.frequency_destiny = frequency_table[system_status.inputs_status];

188system_status_e update_meas(uint16_t vbus_meas, uint16_t ibus_meas) {
189 bool in_emergency = false;
190 system_status.vbus_min = vbus_meas;
191 system_status.ibus_max = ibus_meas;
192
193 if ( system_status.ibus_max > system_seccurity_settings.ibus_max ) {
194 in_emergency = true;
195 if ( system_status.status != SYSTEM_EMERGENCY ) {
196 ESP_LOGE( TAG, "Disparo de emergencia por sobre corriente.");
197 }
198 }
199
200 if ( system_status.vbus_min < system_seccurity_settings.vbus_min ) {
201 in_emergency = true;
202 if ( system_status.status != SYSTEM_EMERGENCY ) {
203 ESP_LOGE( TAG, "Disparo de emergencia por baja tensión.");
204 }
205 }
206
207 if ( in_emergency == true ) {
208 if ( system_status.status == SYSTEM_EMERGENCY ) {
209 SystemEventPost(SECURITY_EXCEDED);
210 ESP_LOGI( TAG, "Mando senal.");
211 }
212 system_status.status = SYSTEM_EMERGENCY_SENT;
213 } else if ( system_status.status == SYSTEM_EMERGENCY_SENT ) {
214 SystemEventPost(SECURITY_OK);
215 }
216
217 return system_status.status;

220void set_frequency_table( uint16_t input_variable, uint16_t freq_regime ) {
221 if ( input_variable == 1 ) { //lineal
222 uint16_t frequency_gap = freq_regime / 8;
223 frequency_table[7] = frequency_gap;
224 for (uint8_t i = 6; i > 0; i-- ) {
225 frequency_table[i] = frequency_table[i + 1] + frequency_gap;
226 ESP_LOGI(TAG, "frequency_table[%d] = %d", i, frequency_table[i]);
227 }
228 frequency_table[0] = freq_regime;
229 ESP_LOGI(TAG, "frequency_table[0] = %d", frequency_table[0]);
230 } else if ( input_variable == 2 ) { //Cuadratica
231 for (uint8_t i = 0; i < 8; i++) {
232 uint32_t num = (uint32_t)freq_regime * (uint32_t)(i + 1) * (uint32_t)(i + 1);
233 uint16_t fi = (uint16_t)((num + 32) / 64);
234 if (fi == 0)
235 fi = 1;
236 frequency_table[7 - i] = fi;
237 ESP_LOGI(TAG, "frequency_table[%d] = %d", 7 - i, frequency_table[7 - i]);
238 }
239 }
240}
