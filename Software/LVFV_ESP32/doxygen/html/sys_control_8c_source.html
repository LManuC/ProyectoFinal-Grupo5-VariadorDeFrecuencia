67static SPI_Response SPI_SendRequest(spi_cmd_item_t *spi_cmd_item) {
68
69 uint8_t tx_buffer[4];
70 uint8_t rx_buffer[4];
71 esp_err_t ret;
72 int flagDevolverValor; // Se necesita devolver el valor por parametro
73
74 ESP_LOGI( TAG, "[SPI Module] Request %d\n", spi_cmd_item->request);
75
76 // Chequeo de errores fuera de rango y comando desconocido
77 if(spi_cmd_item->setValue > 512 || spi_cmd_item->setValue < 0) {
78 return SPI_RESPONSE_ERR_DATA_INVALID;
79 }
80
81 if(spi_cmd_item->request < SPI_REQUEST_START || spi_cmd_item->request > SPI_REQUEST_EMERGENCY) {
82 ESP_LOGI( TAG, "[SPI Module] Comando desconocido\n");
83 return SPI_RESPONSE_ERR_CMD_UNKNOWN;
84 }
85
86 // Es de los comandos que deben devolver un valor por parametro?
87 if(spi_cmd_item->request >= SPI_REQUEST_GET_FREC && spi_cmd_item->request <= SPI_REQUEST_IS_STOP) {
88 flagDevolverValor = 1;
89 }else {
90 flagDevolverValor = 0;
91 }
92
93 // Se arma la cadena a enviar
94 tx_buffer[0] = spi_cmd_item->request;
95
96 if(spi_cmd_item->setValue > 255) {
97 tx_buffer[1] = 255;
98 tx_buffer[2] = spi_cmd_item->setValue - 255;
99 tx_buffer[3] = ';';
100 }else {
101 tx_buffer[1] = spi_cmd_item->setValue;
102 tx_buffer[2] = 0;
103 tx_buffer[3] = ';';
104 }
105
106
107 // Limpiamos el buffer de recepcion
108 memset(rx_buffer, 0, sizeof(rx_buffer));
109
110 // Siempre envio y recibo 4 bytes
111 spi_transaction_t t = {
112 .length = 8 * 4,
113 .tx_buffer = tx_buffer,
114 .rx_buffer = rx_buffer,
115 .rxlength = 8 * 4,
116 };
117
118 ret = spi_device_transmit(spi_handle, &t);
119 if (ret != ESP_OK) {
120 ESP_LOGI( TAG, "[ESP32] Error en SPI transmit: %d\n", ret);
121 return SPI_RESPONSE_ERR;
122 }
123
124 tx_buffer[0] = SPI_REQUEST_RESPONSE;
125 tx_buffer[1] = ';';
126
127 t.length = 8 * 4;
128 t.tx_buffer = tx_buffer;
129 t.rx_buffer = rx_buffer;
130 t.rxlength = 8 * 4;
131
132 vTaskDelay(pdMS_TO_TICKS(200));
133
134 ret = spi_device_transmit(spi_handle, &t);
135 if (ret != ESP_OK) {
136 ESP_LOGI( TAG, "[ESP32] Error en SPI transmit: %d\n", ret);
137 return SPI_RESPONSE_ERR;
138 }
139
140
141
142 ESP_LOGI( TAG, "rx_buffer[0]: %d", rx_buffer[0]);
143 if(flagDevolverValor && rx_buffer[0] == SPI_RESPONSE_OK) {
144
145 spi_cmd_item->getValue = rx_buffer[1] + rx_buffer[2];
146 ESP_LOGI( TAG, "RxValores: %d - %d", rx_buffer[1], rx_buffer[2]);
147
148 }else {
149 spi_cmd_item->getValue = 0;
150 }
151
152 return rx_buffer[0];
153}

168 .spics_io_num = PIN_NUM_CS, // CS pin
169 .queue_size = 1,

171 };
172
173 // Inicializar bus SPI

175 if ( ret != ESP_OK ) {
176 return ret;
177 }
178 ESP_ERROR_CHECK(ret);
179
180 // Añadir dispositivo

224 xQueueSend(system_event_queue, &emergency, 0);
