166static esp_err_t program_alarm(esp_timer_handle_t *timer_handle, void (*callback)(void *), int hour, int minute, const char *name) {
167 time_t now;
168 struct tm timeinfo_alarm;
169 time(&now);
170 localtime_r(&now, &timeinfo_alarm);
171
172 struct tm alarm_tm = timeinfo_alarm;
173 alarm_tm.tm_hour = hour;
174 alarm_tm.tm_min = minute;
175 alarm_tm.tm_sec = 0;
176
177 time_t alarm_epoch = mktime(&alarm_tm);
178 if (alarm_epoch <= now) {
179 alarm_epoch += 24 * 3600;
180 }
181
182 int64_t usec_until_alarm = (alarm_epoch - now) * 1000000LL;
183
184 esp_timer_create_args_t timer_args = {
185 .callback = callback,
186 .arg = NULL,
187 .dispatch_method = ESP_TIMER_TASK,
188 .name = name
189 };
190
191 esp_err_t err = esp_timer_create(&timer_args, timer_handle);
192 if (err != ESP_OK) {
193 return err;
194 }
195
196 err = esp_timer_start_once(*timer_handle, usec_until_alarm);
197 if (err != ESP_OK) {
198 return err;
199 }
200
201 ESP_LOGI(TAG, "Programada %s para %02d:%02d (en %lld segundos)", name, hour, minute, (alarm_epoch - now));
202
203 return ESP_OK;
204}
