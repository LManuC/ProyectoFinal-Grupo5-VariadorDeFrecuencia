43static void SPI_ProcesarComando(uint8_t* buffer, int cantBytes, uint8_t* bufferResponse) {
44 int resp;
45 int val;
46
47 // Selección por comando (primer byte del buffer de respuesta
48 bufferResponse[0] = '\0';
49 switch (buffer[0]) {
50 case SPI_REQUEST_START:
51 resp = GestorEstados_Action(ACTION_START, 0);
52
53 if(resp == ACTION_RESP_OK) {
54 bufferResponse[0] = SPI_RESPONSE_OK;
55 } else if(resp == ACTION_RESP_MOVING) {
56 bufferResponse[0] = SPI_RESPONSE_ERR_MOVING;
57 } else if(resp == ACTION_RESP_EMERGENCY_ACTIVE) {
58 bufferResponse[0] = SPI_RESPONSE_ERR_EMERGENCY_ACTIVE;
59 } else {
60 bufferResponse[0] = SPI_RESPONSE_ERR;
61 }
62
63 bufferResponse[1] = ';';
64 return;
65
66 case SPI_REQUEST_STOP:
67 resp = GestorEstados_Action(ACTION_STOP, 0);
68
69 if(resp == ACTION_RESP_OK) {
70 bufferResponse[0] = SPI_RESPONSE_OK;
71 } else if(resp == ACTION_RESP_NOT_MOVING) {
72 bufferResponse[0] = SPI_RESPONSE_ERR_NOT_MOVING;
73 } else {
74 bufferResponse[0] = SPI_RESPONSE_ERR;
75 }
76
77 bufferResponse[1] = ';';
78 return;
79
80 case SPI_REQUEST_EMERGENCY:
81 GestorEstados_Action(ACTION_EMERGENCY, 0);
82 bufferResponse[0] = SPI_RESPONSE_OK;
83 bufferResponse[1] = ';';
84 return;
85
86 case SPI_REQUEST_SET_FREC:
87 val = (uint8_t)buffer[1] + (uint8_t)buffer[2];
88 resp = GestorEstados_Action(ACTION_SET_FREC, val);
89
90 if(resp == ACTION_RESP_OK) {
91 bufferResponse[0] = SPI_RESPONSE_OK;
92 } else if(resp == ACTION_RESP_OUT_RANGE) {
93 bufferResponse[0] = SPI_RESPONSE_ERR_DATA_OUT_RANGE;
94 } else if(resp == ACTION_RESP_MOVING) {
95 bufferResponse[0] = SPI_RESPONSE_ERR_MOVING;
96 } else {
97 bufferResponse[0] = SPI_RESPONSE_ERR;
98 }
99
100 bufferResponse[1] = ';';
101 return;
102
103 case SPI_REQUEST_SET_ACEL:
104 resp = GestorEstados_Action(ACTION_SET_ACEL, (uint8_t)buffer[1]);
105
106 if(resp == ACTION_RESP_OK) {
107 bufferResponse[0] = SPI_RESPONSE_OK;
108 } else if(resp == ACTION_RESP_OUT_RANGE) {
109 bufferResponse[0] = SPI_RESPONSE_ERR_DATA_OUT_RANGE;
110 } else if(resp == ACTION_RESP_MOVING) {
111 bufferResponse[0] = SPI_RESPONSE_ERR_MOVING;
112 } else {
113 bufferResponse[0] = SPI_RESPONSE_ERR;
114 }
115
116 bufferResponse[1] = ';';
117 return;
118
119 case SPI_REQUEST_SET_DESACEL:
120 resp = GestorEstados_Action(ACTION_SET_DESACEL, (uint8_t)buffer[1]);
121
122 if(resp == ACTION_RESP_OK) {
123 bufferResponse[0] = SPI_RESPONSE_OK;
124 } else if(resp == ACTION_RESP_OUT_RANGE) {
125 bufferResponse[0] = SPI_RESPONSE_ERR_DATA_OUT_RANGE;
126 } else if(resp == ACTION_RESP_MOVING) {
127 bufferResponse[0] = SPI_RESPONSE_ERR_MOVING;
128 } else {
129 bufferResponse[0] = SPI_RESPONSE_ERR;
130 }
131
132 bufferResponse[1] = ';';
133 return;
134
135 case SPI_REQUEST_SET_DIR:
136 val = (uint8_t)buffer[1];
137 resp = GestorEstados_Action(ACTION_SET_DIR, val);
138
139 if(resp == ACTION_RESP_OK) {
140 bufferResponse[0] = SPI_RESPONSE_OK;
141 } else if(resp == ACTION_RESP_OUT_RANGE) {
142 bufferResponse[0] = SPI_RESPONSE_ERR_DATA_OUT_RANGE;
143 } else if(resp == ACTION_RESP_MOVING) {
144 bufferResponse[0] = SPI_RESPONSE_ERR_MOVING;
145 } else {
146 bufferResponse[0] = SPI_RESPONSE_ERR;
147 }
148
149 bufferResponse[1] = ';';
150 return;
151
152 case SPI_REQUEST_GET_FREC:
153 /* Lee valor actual de frecuencia desde el SVM */
154 val = GestorSVM_GetFrec();
155 bufferResponse[0] = SPI_RESPONSE_OK;
156 /* Empaquetado simple en 2 bytes (LOW/HIGH “capado”) */
157 if (val > 255) {
158 bufferResponse[1] = 255;
159 bufferResponse[2] = (uint8_t)(val - 255);
160 } else {
161 bufferResponse[1] = (uint8_t)val;
162 bufferResponse[2] = 0;
163 }
164 bufferResponse[3] = ';';
165 return;
166
167 case SPI_REQUEST_GET_ACEL:
168 bufferResponse[0] = SPI_RESPONSE_OK;
169 bufferResponse[1] = GestorSVM_GetAcel();
170 bufferResponse[2] = 0;
171 bufferResponse[3] = ';';
172 return;
173
174 case SPI_REQUEST_GET_DESACEL:
175 bufferResponse[0] = SPI_RESPONSE_OK;
176 bufferResponse[1] = GestorSVM_GetDesacel();
177 bufferResponse[2] = 0;
178 bufferResponse[3] = ';';
179 return;
180
181 case SPI_REQUEST_GET_DIR:
182 bufferResponse[0] = SPI_RESPONSE_OK;
183 bufferResponse[1] = GestorSVM_GetDir();
184 bufferResponse[2] = 0;
185 bufferResponse[3] = ';';
186 return;
187
188 case SPI_REQUEST_IS_STOP:
189 bufferResponse[0] = SPI_RESPONSE_OK;
190 bufferResponse[1] = (uint8_t)GestorEstados_Action(ACTION_IS_MOTOR_STOP, 0);
191 bufferResponse[2] = 0;
192 bufferResponse[3] = ';';
193 return;
194
195 case SPI_REQUEST_RESPONSE:
196 bufferResponse[0] = SPI_RESPONSE_OK;
197 bufferResponse[1] = ';';
198 return;
199
200 default:
201 /* Comando desconocido */
202 bufferResponse[0] = SPI_RESPONSE_ERR_CMD_UNKNOWN;
203 bufferResponse[1] = ';';
204 return;
205 }
206
207 /* Fallback (no debería alcanzarse) */
208 bufferResponse[0] = SPI_RESPONSE_ERR;
209 bufferResponse[1] = ';';
210}

223void HAL_SPI_TxRxCpltCallback(SPI_HandleTypeDef *_hspi) {
224 int len = 0;
225 int found = 0;
226
227 if (_hspi->Instance == SPI2) {
228 /* Buscar fin de comando ';' en el buffer de RX */
229 for (int i = 0; i < SPI_BUF_SIZE; i++) {
230 if (rxDMABuffer[i] == ';') {
231 found = 1;
232 len = i; // bytes válidos (sin incluir ';')
233 break;
234 }
235 }
236
237 /* Respuesta por defecto: “ERR_NO_COMMAND;” */
238 uint8_t resp[4] = { SPI_RESPONSE_ERR_NO_COMMAND, ';', 0, 0 };
239
240 if (found && len > 0) {
241 SPI_ProcesarComando(rxDMABuffer, len, resp);
242 }
243
244 txDMABuffer[0] = resp[0];
245 txDMABuffer[1] = resp[1];
246 txDMABuffer[2] = resp[2];
247 txDMABuffer[3] = resp[3];
248
249 /* Limpiar RX para próxima captura */
250 memset(rxDMABuffer, 0, sizeof(rxDMABuffer));
251
252 /* Reiniciar ciclo DMA full-duplex (no bloqueante) */
253 HAL_SPI_TransmitReceive_DMA(hspi, txDMABuffer, rxDMABuffer, SPI_TRANSMITION_SIZE);
254 }

257void SPI_Init(void* _hspi) {
258 /* Prioridades/enable de DMA para SPI2: TX (CH4), RX (CH5) */
259 HAL_NVIC_SetPriority(DMA1_Channel4_IRQn, 2, 0);
260 HAL_NVIC_EnableIRQ(DMA1_Channel4_IRQn);
261
262 HAL_NVIC_SetPriority(DMA1_Channel5_IRQn, 2, 0);
263 HAL_NVIC_EnableIRQ(DMA1_Channel5_IRQn);
264
265 /* Guardar handle de SPI */
266 hspi = (SPI_HandleTypeDef*)_hspi;
267
268 /* Inicialización básica de buffers */
269 memset(rxDMABuffer, 0, sizeof(rxDMABuffer));
270 memset(txDMABuffer, 0, sizeof(txDMABuffer));
271
272 /* Respuesta inicial por defecto: “OK;” (se enviará en la primera TxRx) */
273 txDMABuffer[0] = SPI_RESPONSE_OK;
274 txDMABuffer[1] = ';';
275
276 /* Iniciar ciclo DMA full-duplex no bloqueante */
277 HAL_SPI_TransmitReceive_DMA(hspi, txDMABuffer, rxDMABuffer, SPI_TRANSMITION_SIZE);
278}
