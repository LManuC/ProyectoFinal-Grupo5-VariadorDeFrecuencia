223static int pinMap(int x) {
224 int r = 0;
225 x &= 0x07; // limitar a 3 bits (U,V,W)
226 if (x == 1) {
227 r = 2; // W
228 }
229 else if (x == 2) {
230 r = 1; // V
231 }
232 else if (x == 4) {
233 r = 0; // U
234 }
235 return r;

238static void GestorSVM_CalcularValoresSwitching() {
239 int t1, t2, t0;
240 int ticksC1, ticksC2, ticksC3;
241
242 /* Rampa de velocidad si corresponde */
243 if (flagChangingFrecuencia) {
244 GestorSVM_Calculoaceleracioneracion();
245 }
246
247 /* Avance angular y gestión de diente 0..60° */
248 anguloActual += anguloSwitching;
249 if (anguloActual >= 360 * 1000 * 1000) {
250 anguloActual -= 360 * 1000 * 1000;
251 cuadranteActual = 0;
252 flagAscensoAnguloParcial = 1;
253 anguloParcial = anguloActual;
254 } else {
255 if (flagAscensoAnguloParcial == 1) {
256 anguloParcial += anguloSwitching;
257 if (anguloParcial >= 60 * 1000 * 1000) {
258 anguloParcial = 2 * 60 * 1000 * 1000 - anguloParcial; // espejo
259 flagAscensoAnguloParcial = 0;
260 cuadranteActual = (cuadranteActual + 1) % 6;
261 }
262 } else {
263 anguloParcial -= anguloSwitching;
264 if (anguloParcial <= 0) {
265 anguloParcial = -anguloParcial; // espejo
266 flagAscensoAnguloParcial = 1;
267 cuadranteActual = (cuadranteActual + 1) % 6;
268 }
269 }
270 }
271
272 /* t1/t2 por regresión e índice de modulación */
273 t1 = (int)(((CONST_CALC_T1_PROP * (float)anguloParcial + CONST_CALC_T1_ORD_ORG) * indiceModulacion) / 100);
274 t2 = (int)(((CONST_CALC_T2_PROP * (float)anguloParcial + CONST_CALC_T2_ORD_ORG) * indiceModulacion) / 100);
275 t0 = (int)((float)(255 - t1 - t2) * 0.5);
276
277 /* Ticks absolutos en el período */
278 ticksC1 = t0;
279 ticksC2 = t0 + t1;
280 ticksC3 = t0 + t1 + t2;
281
282 /* Clasificación de interferencias */
283 if (ticksC3 - ticksC1 < MIN_TICKS_DIF) {
284 casoInterferencia = INTERF_T1T3;
285 } else if (ticksC2 - ticksC1 < MIN_TICKS_DIF && ticksC3 - ticksC2 < MIN_TICKS_DIF) {
286 casoInterferencia = INTERF_T1T2_T2T3;
287 } else if (ticksC2 - ticksC1 < MIN_TICKS_DIF) {
288 casoInterferencia = INTERF_T1T2;
289 } else if (ticksC3 - ticksC2 < MIN_TICKS_DIF) {
290 casoInterferencia = INTERF_T2T3;
291 } else {
292 casoInterferencia = INTERF_NULA;
293 }
294
295 ticksChannel[0] = ticksC1;
296 ticksChannel[1] = ticksC2;
297 ticksChannel[2] = ticksC3;
298}

300static void GestorSVM_SwitchInterrupt(SwitchInterruptType intType) {
301 volatile static int countUpTx[3]; // 1=subiendo, 0=bajando
302 volatile static int flagTxActivo[3]; // espera de evento por canal
303 volatile static CasoInterferenciaTimer interferencia;
304 volatile static int cuadrante = 0;
305 int ticks1, ticks2, ticks3;
306
307 /* Si no hay datos precargados, no hacer nada */
308 if (bufferCalculo.contadorDeDatos <= 0) {
309 return;
310 }
311
312 switch (intType) {
313 case SWITCH_INT_CH1:
314 if (flagTxActivo[0]) {
315 flagTxActivo[0] = 0;
316 if (countUpTx[0]) {
317 GestorSVM_SwitchPuertos(ORDEN_SWITCH_1_UP, 1, cuadrante);
318 countUpTx[0] = 0;
319 } else {
320 countUpTx[0] = 1;
321 GestorSVM_SwitchPuertos(ORDEN_SWITCH_1_DOWN, 0, cuadrante);
322 /* Rehabilita interrupciones de los otros canales */
323 TIM3->DIER |= TIM_DIER_CC2IE;
324 TIM3->DIER |= TIM_DIER_CC3IE;
325 TIM3->DIER |= TIM_DIER_CC4IE;
326 }
327 } else {
328 flagTxActivo[0] = 1;
329 }
330 break;
331
332 case SWITCH_INT_CH2:
333 if (flagTxActivo[1]) {
334 flagTxActivo[1] = 0;
335 if (countUpTx[1]) {
336 GestorSVM_SwitchPuertos(ORDEN_SWITCH_2_UP, 1, cuadrante);
337 countUpTx[1] = 0;
338 } else {
339 GestorSVM_SwitchPuertos(ORDEN_SWITCH_2_DOWN, 0, cuadrante);
340 countUpTx[1] = 1;
341 }
342 } else {
343 if (interferencia == INTERF_T1T2_T2T3) {
344 flagTxActivo[1] = 0;
345 TIM3->DIER &= ~TIM_DIER_CC3IE; // desactiva T2
346 } else {
347 flagTxActivo[1] = 1;
348 }
349 }
350 break;
351
352 case SWITCH_INT_CH3:
353 if (flagTxActivo[2]) {
354 flagTxActivo[2] = 0;
355 if (countUpTx[2]) {
356 GestorSVM_SwitchPuertos(ORDEN_SWITCH_3_UP, 1, cuadrante);
357
358 /* Manejo de interferencias */
359 switch (interferencia) {
360 case INTERF_NULA:
361 flagTxActivo[0] = 1;
362 flagTxActivo[1] = 1;
363 flagTxActivo[2] = 1;
364 break;
365 case INTERF_T1T2:
366 flagTxActivo[0] = 0; // T1 pendiente
367 flagTxActivo[1] = 0; // desactiva T2
368 flagTxActivo[2] = 1;
369 // Deshabilitacion T2
370 TIM3->DIER &= ~TIM_DIER_CC3IE;
371 // Activacion T1
372 TIM3->DIER |= TIM_DIER_CC2IE;
373 break;
374 case INTERF_T2T3:
375 flagTxActivo[0] = 1;
376 flagTxActivo[1] = 0; // T2 pendiente
377 flagTxActivo[2] = 0; // T3 off
378 TIM3->DIER &= ~TIM_DIER_CC4IE; // CC4 off
379 TIM3->DIER |= TIM_DIER_CC3IE; // CC3 on
380 break;
381 case INTERF_T1T3:
382 /* No debe ocurrir aquí */
383 break;
384 case INTERF_T1T2_T2T3:
385 flagTxActivo[0] = 0;
386 flagTxActivo[1] = 0;
387 flagTxActivo[2] = 1;
388 TIM3->DIER &= ~TIM_DIER_CC3IE; // CC3 off
389 TIM3->DIER |= TIM_DIER_CC2IE; // CC2 on
390 break;
391 }
392 /* Próximas interrupciones: cuenta abajo */
393 countUpTx[0] = 0;
394 countUpTx[1] = 0;
395 countUpTx[2] = 0;
396 } else {
397 GestorSVM_SwitchPuertos(ORDEN_SWITCH_3_DOWN, 0, cuadrante);
398 countUpTx[2] = 1;
399 }
400 } else {
401 if (interferencia == INTERF_T1T3) {
402 flagTxActivo[0] = 0;
403 flagTxActivo[1] = 0;
404 flagTxActivo[2] = 0;
405 } else {
406 flagTxActivo[2] = 1;
407 }
408 }
409 break;
410
411 case SWITCH_INT_RESET:
412 /* Re-sync / precarga */
413 countUpTx[0] = 1;
414 countUpTx[1] = 1;
415 countUpTx[2] = 1;
416
417 /* Consume entrada del buffer */
418 ticks1 = bufferCalculo.ticksChannel1[bufferCalculo.indiceLectura];
419 ticks2 = bufferCalculo.ticksChannel2[bufferCalculo.indiceLectura];
420 ticks3 = bufferCalculo.ticksChannel3[bufferCalculo.indiceLectura];
421 interferencia = bufferCalculo.casoInterferencia[bufferCalculo.indiceLectura];
422 cuadrante = bufferCalculo.cuadranteActual[bufferCalculo.indiceLectura];
423 bufferCalculo.indiceLectura = (bufferCalculo.indiceLectura + 1) % BUFFER_CALCULO_SIZE;
424 bufferCalculo.contadorDeDatos--;
425
426 /* Habilitación según interferencias */
427 switch (interferencia) {
428 case INTERF_NULA:
429 flagTxActivo[0] = 1;
430 flagTxActivo[1] = 1;
431 flagTxActivo[2] = 1;
432 TIM3->DIER |= TIM_DIER_CC2IE;
433 TIM3->DIER |= TIM_DIER_CC3IE;
434 TIM3->DIER |= TIM_DIER_CC4IE;
435 break;
436 case INTERF_T1T2:
437 flagTxActivo[0] = 0;
438 flagTxActivo[1] = 1;
439 flagTxActivo[2] = 1;
440 TIM3->DIER &= ~TIM_DIER_CC2IE;
441 TIM3->DIER |= TIM_DIER_CC3IE;
442 TIM3->DIER |= TIM_DIER_CC4IE;
443 break;
444 case INTERF_T2T3:
445 flagTxActivo[0] = 1;
446 flagTxActivo[1] = 0;
447 flagTxActivo[2] = 1;
448 TIM3->DIER |= TIM_DIER_CC2IE;
449 TIM3->DIER &= ~TIM_DIER_CC3IE;
450 TIM3->DIER |= TIM_DIER_CC4IE;
451 break;
452 case INTERF_T1T3:
453 flagTxActivo[0] = 0;
454 flagTxActivo[1] = 0;
455 flagTxActivo[2] = 0;
456 /* Carga posiciones distantes (nunca disparan simultáneo) */
457 ticks1 = 20;
458 ticks2 = 60;
459 ticks3 = 100;
460 TIM3->DIER |= TIM_DIER_CC2IE;
461 TIM3->DIER |= TIM_DIER_CC3IE;
462 TIM3->DIER |= TIM_DIER_CC4IE;
463 break;
464 case INTERF_T1T2_T2T3:
465 // Desactivacion de T1 y Activacion de T2 y T3
466 flagTxActivo[0] = 0;
467 flagTxActivo[1] = 1;
468 flagTxActivo[2] = 1;
469 TIM3->DIER &= ~TIM_DIER_CC2IE;
470 TIM3->DIER |= TIM_DIER_CC3IE;
471 TIM3->DIER |= TIM_DIER_CC4IE;
472 break;
473 }
474
475 /* Precarga CCRs (t1→CCR2, t2→CCR3, t3→CCR4) */
476 TIM3->CCR2 = ticks1;
477 TIM3->CCR3 = ticks2;
478 TIM3->CCR4 = ticks3;
479 break;
480
481 case SWITCH_INT_CLEAN:
482 /* Limpieza de estado */
483 countUpTx[0] = 1;
484 countUpTx[1] = 1;
485 countUpTx[2] = 1;
486 flagTxActivo[0] = 0;
487 flagTxActivo[1] = 0;
488 flagTxActivo[2] = 0;
489 break;
490
491 default:
492 break;
493 }
494}

513static void GestorSVM_Calculoaceleracioneracion() {
514 /* Sincronización con parámetros sombra, si corresponde */
515 if (flagActualizarParamSombra) {
516 frecObjetivo = paramSombra.frecObjetivo;
517 cambioFrecuenciaPorCiclo = paramSombra.cambioFrecuenciaPorCiclo;
518 direccionRotacion = paramSombra.direccionRotacion;
519 flagMotorRunning = paramSombra.flagMotorRunning;
520 flagEsAcelerado = paramSombra.flagEsAcelerado;
521 flagChangingFrecuencia = paramSombra.flagChangingFrecuencia;
522 flagActualizarParamSombra = 0;
523 }
524
525 /* Aplicación de rampa */
526 if (flagEsAcelerado) {
527 frecuenciaSalida += cambioFrecuenciaPorCiclo;
528 if (frecuenciaSalida >= frecObjetivo) {
529 frecuenciaSalida = frecObjetivo;
530 GestorEstados_Action(ACTION_TO_CONST_RUNNING, 0);
531 flagChangingFrecuencia = 0;
532 }
533 } else {
534 frecuenciaSalida -= cambioFrecuenciaPorCiclo;
535 if (frecuenciaSalida <= frecObjetivo) {
536 frecuenciaSalida = frecObjetivo;
537
538 if (frecObjetivo == 0) {
539 flagMotorRunning = 0;
540 GestorTimers_DetenerTimerSVM();
541 GestorSVM_SwitchInterrupt(SWITCH_INT_CLEAN);
542
543 /* Limpia buffer productor/consumidor */
544 bufferCalculo.indiceEscritura = 0;
545 bufferCalculo.indiceLectura = 0;
546 bufferCalculo.contadorDeDatos = 0;
547
548 /* Apaga salidas y drivers */
549 HAL_GPIO_WritePin(GPIOA, GPIO_U_IN, GPIO_PIN_RESET);
550 HAL_GPIO_WritePin(GPIOA, GPIO_V_IN, GPIO_PIN_RESET);
551 HAL_GPIO_WritePin(GPIOA, GPIO_W_IN, GPIO_PIN_RESET);
552
553 HAL_GPIO_WritePin(GPIOA, GPIO_U_SD, GPIO_PIN_RESET);
554 HAL_GPIO_WritePin(GPIOA, GPIO_V_SD, GPIO_PIN_RESET);
555 HAL_GPIO_WritePin(GPIOA, GPIO_W_SD, GPIO_PIN_RESET);
556
557 /* Notifica detención */
558 GestorEstados_Action(ACTION_MOTOR_STOPPED, 0);
559 } else {
560 GestorEstados_Action(ACTION_TO_CONST_RUNNING, 0);
561 }
562 flagChangingFrecuencia = 0;
563 }
564 }
565
566 /* Índice de modulación (V/f) */
567 if (frecuenciaSalida < 50 * 1000 * 1000) {
568 indiceModulacion = frecuenciaSalida / (500 * 1000); // lineal hasta 50 Hz
569 } else {
570 indiceModulacion = 100;
571 }
572 if (indiceModulacion < 1) {
573 indiceModulacion = 1;
574 }
575
576 /* Incremento angular por switching (cuidar orden para evitar overflow) */
577 anguloSwitching = (frecuenciaSalida / frecuenciaSwitching) * 360;
578}

585void GestorSVM_SetConfiguration(ConfiguracionSVM* configuracion) {
586 int i;
587 /* Dinámicos */
588 frecuenciaSwitching = configuracion->frecuenciaSwitching;
589 direccionRotacion = configuracion->direccionRotacion;
590 aceleracion = configuracion->aceleracion;
591 desaceleracion = configuracion->desaceleracion;
592
593 /* Referencia (inicia como target) */
594 GestorSVM_SetFrec(configuracion->frecuenciaReferenica);
595
596 printf("Configuracion Seteada \n");
597 int estado0, estado1, estado2, estado3;
598 int pinToggle1, pinToggle2, pinToggle3;
599 uint32_t myBSRR;
600
601 estadoGPIOOff = (GPIO_U_IN | GPIO_V_IN | GPIO_W_IN) << 16;
602 estadoGPIOOn = (GPIO_U_IN | GPIO_V_IN | GPIO_W_IN);
603
604 for (i = 0; i < 6; i++) {
605 estado0 = vectorSecuenciaPorCuadrante[i][0];
606 estado1 = vectorSecuenciaPorCuadrante[i][1];
607 estado2 = vectorSecuenciaPorCuadrante[i][2];
608 estado3 = vectorSecuenciaPorCuadrante[i][3];
609
610 /* Estado 0→1 */
611 myBSRR = 0;
612 myBSRR |= ((estado1 & 0b100) > 0) ? puerto_senal_pierna[0] : (puerto_senal_pierna[0] << 16);
613 myBSRR |= ((estado1 & 0b010) > 0) ? puerto_senal_pierna[1] : (puerto_senal_pierna[1] << 16);
614 myBSRR |= ((estado1 & 0b001) > 0) ? puerto_senal_pierna[2] : (puerto_senal_pierna[2] << 16);
615 estadoGPIOPorCuadranteYOrden[i][0] = myBSRR;
616
617 /* Estado 1→2 */
618 myBSRR = 0;
619 myBSRR |= ((estado2 & 0b100) > 0) ? puerto_senal_pierna[0] : (puerto_senal_pierna[0] << 16);
620 myBSRR |= ((estado2 & 0b010) > 0) ? puerto_senal_pierna[1] : (puerto_senal_pierna[1] << 16);
621 myBSRR |= ((estado2 & 0b001) > 0) ? puerto_senal_pierna[2] : (puerto_senal_pierna[2] << 16);
622 estadoGPIOPorCuadranteYOrden[i][1] = myBSRR;
623
624 /* Pines que conmutan entre estados (XOR) */
625 pinToggle1 = pinMap(estado0 ^ estado1);
626 pinToggle2 = pinMap(estado1 ^ estado2);
627 pinToggle3 = pinMap(estado2 ^ estado3);
628
629 /* Mapa a U/V/W con desplazamiento */
630 pinTogglePorCuadranteYOrden[i][0] = GPIO_U_IN << (pinToggle1 * 2);
631 pinTogglePorCuadranteYOrden[i][1] = GPIO_U_IN << (pinToggle2 * 2);
632 pinTogglePorCuadranteYOrden[i][2] = GPIO_U_IN << (pinToggle3 * 2);
633 }

668int GestorSVM_SetFrec(int frec) {
669 int flagEsAcelerado_local;
670 int32_t cambioFrecuenciaPorCiclo_local;
671 int32_t frecTarget_local;
672 int32_t nuevaFrec;
673
674 if (frec < FERC_OUT_MIN || frec > FERC_OUT_MAX) {
675 return -1;
676 }
677 nuevaFrec = (int32_t)frec * 1000 * 1000;
678 if (frecuenciaSalida == nuevaFrec) {
679 return -2;
680 }
681
682 if (flagMotorRunning) {
683 /* Determinar sentido de la rampa */
684 if (frecuenciaSalida < nuevaFrec) {
685 flagEsAcelerado_local = 1;
686 cambioFrecuenciaPorCiclo_local = (aceleracion * 1000 * 1000) / (frecuenciaSwitching);
687 } else {
688 flagEsAcelerado_local = 0;
689 cambioFrecuenciaPorCiclo_local = (desaceleracion * 1000 * 1000) / (frecuenciaSwitching);
690 }
691
692 frecTarget_local = nuevaFrec;
693 frecuenciaReferenica = frec;
694
695 /* Parámetros sombra */
696 paramSombra.flagMotorRunning = 1;
697 paramSombra.flagChangingFrecuencia = 1;
698 paramSombra.flagEsAcelerado = flagEsAcelerado_local;
699 paramSombra.cambioFrecuenciaPorCiclo = cambioFrecuenciaPorCiclo_local;
700 paramSombra.frecObjetivo = frecTarget_local;
701 flagActualizarParamSombra = 1;
702
703 flagChangingFrecuencia = 1;
704 return 1;
705 } else {
706 frecuenciaReferenica = frec;
707 return 0;
708 }
709}

717int GestorSVM_SetDir(int dir) {
718 int estado1, estado2;
719 int i;
720 uint32_t myBSRR;
721
722 if (flagMotorRunning) {
723 return -2;
724 }
725 if (flagChangingFrecuencia) {
726 return -2;
727 }
728 if (dir != 0 && dir != 1) {
729 return -1;
730 }
731 if (dir == direccionRotacion) {
732 return 0;
733 }
734
735 /* Recalcular tablas BSRR intercambiando U↔V */

738 estado2 = vectorSecuenciaPorCuadrante[i][2];
739

741 myBSRR |= ((estado1 & 0b100) ? GPIO_V_IN : (GPIO_V_IN << 16));
742 myBSRR |= ((estado1 & 0b010) ? GPIO_U_IN : (GPIO_U_IN << 16));
743 myBSRR |= ((estado1 & 0b001) ? GPIO_W_IN : (GPIO_W_IN << 16));
744 estadoGPIOPorCuadranteYOrden[i][0] = myBSRR;
745
746 myBSRR = 0;
747 myBSRR |= ((estado2 & 0b100) ? GPIO_V_IN : (GPIO_V_IN << 16));
748 myBSRR |= ((estado2 & 0b010) ? GPIO_U_IN : (GPIO_U_IN << 16));
749 myBSRR |= ((estado2 & 0b001) ? GPIO_W_IN : (GPIO_W_IN << 16));

751 }
